### 1. Two Pointers (–¥–≤–∞ —É–∫–∞–∑–∞—Ç–µ–ª—è)

```go
// –ü–æ–∏—Å–∫ –ø–∞—Ä—ã —á–∏—Å–µ–ª, –¥–∞—é—â–µ–π –Ω—É–∂–Ω—É—é —Å—É–º–º—É –≤ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–º –º–∞—Å—Å–∏–≤–µ
func twoSum(nums []int, target int) []int {
    left, right := 0, len(nums)-1

    for left < right {
        sum := nums[left] + nums[right]
        if sum == target {
            return []int{left, right}
        } else if sum < target {
            left++
        } else {
            right--
        }
    }
    return nil
}
```

### 2. Sliding Window (—Å–∫–æ–ª—å–∑—è—â–µ–µ –æ–∫–Ω–æ)

```go
func maxSumSubarray(nums []int, k int) int {
    if len(nums) < k {
        return 0
    }

    // –°—É–º–º–∞ –ø–µ—Ä–≤–æ–≥–æ –æ–∫–Ω–∞

    for i := k; i < len(nums); i++ {
        windowSum = windowSum - nums[i-k] + nums[i]
        if windowSum > maxSum {
            maxSum = windowSum
        }
    }

    return maxSum
}
```
